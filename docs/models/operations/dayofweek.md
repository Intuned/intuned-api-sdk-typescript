# DayOfWeek

Days of week, one of SUNDAY, MONDAY, TUESDAY, WEDNESDAY, THURSDAY, FRIDAY, SATURDAY


## Supported Types

### `operations.One`

```typescript
const value: operations.One = "SATURDAY";
```

### `operations.DayOfWeek2`

```typescript
const value: operations.DayOfWeek2 = {
  start: "TUESDAY",
  step: 955778,
  end: "MONDAY",
};
```

### `operations.DayOfWeek3`

```typescript
const value: operations.DayOfWeek3 = {
  start: "THURSDAY",
};
```

### `operations.DayOfWeek4[]`

```typescript
const value: operations.DayOfWeek4[] = [
  {
    start: "WEDNESDAY",
  },
];
```

### `operations.DayOfWeek5`

```typescript
const value: operations.DayOfWeek5 = "*";
```
